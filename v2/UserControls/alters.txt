-- Default constraint'leri otomatik bul ve sil
DECLARE @sql NVARCHAR(MAX) = '';
SELECT @sql += 'ALTER TABLE [dbo].[StockMovement] DROP CONSTRAINT ' + QUOTENAME(dc.name) + '; '
FROM sys.default_constraints dc
JOIN sys.columns c ON dc.parent_object_id = c.object_id AND dc.parent_column_id = c.column_id
WHERE dc.parent_object_id = OBJECT_ID('StockMovement')
AND c.name IN ('DeltaKg', 'DeltaMeter', 'DeltaPiece', 'BeforeKg', 'AfterKg', 'BeforeMeter', 'AfterMeter', 'BeforePiece', 'AfterPiece', 'MaterialId');

IF LEN(@sql) > 0 EXEC sp_executesql @sql;
GO

-- Index varsa sil
IF EXISTS (SELECT 1 FROM sys.indexes WHERE name = 'IX_StockMovement_Document' AND object_id = OBJECT_ID('StockMovement'))
    DROP INDEX [IX_StockMovement_Document] ON [dbo].[StockMovement];
GO
